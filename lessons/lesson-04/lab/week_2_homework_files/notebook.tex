    \subsubsection{Features of the dataset}\label{features-of-the-dataset}

    \begin{itemize}
\tightlist
\item
  Cylinders: Number of cylinders of the vehicles engine
\item
  Displacement: Volume of all the cylinders, measured in cubic inches
\item
  Horsepower: Total energy / power developed by the engine
\item
  Weight: Total weight of the vehicle measured in pounds
\item
  Acceleration: Ability of the vehicle to gain speed with respect to
  time
\item
  Model\_Year: Year the vehicle was manufactured
\item
  Origin: Location where the vehicle was manufactured
\item
  Model: Make and model of vehicle
\end{itemize}

    \subsubsection{Response of the dataset}\label{response-of-the-dataset}

    \begin{itemize}
\tightlist
\item
  MPG: Miles Per Gallon, distance the vehicle travels in miles (1.61 km)
  for every gallon (3.78 litres) of fuel consumed
\end{itemize}
